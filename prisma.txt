/**
 * =========================
 * DOCUMENT TEMPLATE

model DocumentTemplate {
  id String @id @default(auto()) @map("_id") @db.ObjectId

  companyId String?  @db.ObjectId
  company   Company? @relation(fields: [companyId], references: [id])

  categoryId String           @db.ObjectId

  documentTypeId String       @db.ObjectId
  documentType   DocumentType @relation(fields: [documentTypeId], references: [id])

  name        String
  slug        String?
  description String?

  templateMode TemplateMode @default(FORM)

  /**
   * layoutJson ใช้เก็บตำแหน่งกล่องข้อความ รูปภาพ ตาราง ลายเซ็น
   * เหมาะกับระบบ drag & drop designer

   * ถ้าต้องการ render จาก HTML/CSS

  paperSize   PaperSize        @default(A4)

  version  Int     @default(1)
  isGlobal Boolean @default(false)
  isActive Boolean @default(true)

  createdByUserId String?      @db.ObjectId
  createdByUser   CompanyUser? @relation("UserCreatedTemplates", fields: [createdByUserId], references: [id])

  createdByAdminId String?      @db.ObjectId
  createdByAdmin   SystemAdmin? @relation("AdminCreatedTemplates", fields: [createdByAdminId], references: [id])

  fields    TemplateField[]
  documents Document[]

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([companyId])
  @@index([categoryId])
  @@index([documentTypeId])
  @@index([isGlobal])
  @@index([isActive])
}

/**
 * =========================
 * DOCUMENT
 * เอกสารที่บริษัทสร้างจริง
 * =========================
 */

model Document {
  id String @id @default(auto()) @map("_id") @db.ObjectId

  companyId String  @db.ObjectId
  company   Company @relation(fields: [companyId], references: [id])

  templateId String?           @db.ObjectId
  template   DocumentTemplate? @relation(fields: [templateId], references: [id])

  categoryId String           @db.ObjectId
  category   DocumentCategory @relation(fields: [categoryId], references: [id])

  documentTypeId String       @db.ObjectId
  documentType   DocumentType @relation(fields: [documentTypeId], references: [id])

  customerId String?          @db.ObjectId
  customer   BusinessPartner? @relation(fields: [customerId], references: [id])

  employeeId String?   @db.ObjectId
  employee   Employee? @relation(fields: [employeeId], references: [id])

  documentNo String
  title      String
  status     DocumentStatus @default(DRAFT)

  /**
   * dataJson คือข้อมูลจริงที่กรอกในฟอร์ม
   * เช่น ชื่อลูกค้า รายการสินค้า วันที่ จำนวนเงิน ลายเซ็น
   */
  dataJson Json

  /**
   * แนะนำเก็บเงินเป็นสตางค์เพื่อไม่ให้เลขเพี้ยน
   */
  subtotalSatang Int?
  discountSatang Int?
  vatSatang      Int?
  totalSatang    Int?

  pdfUrl      String?
  previewUrl  String?
  note        String?

  currentApprovalStep Int?
  rejectedReason      String?

  isLocked Boolean @default(false)

  createdById String      @db.ObjectId
  createdBy   CompanyUser @relation("DocumentCreatedBy", fields: [createdById], references: [id])

  approvedById String?      @db.ObjectId
  approvedBy   CompanyUser? @relation("DocumentApprovedBy", fields: [approvedById], references: [id])

  approvedAt DateTime?
  rejectedAt DateTime?
  cancelledAt DateTime?

  files     DocumentFile[]
  approvals DocumentApproval[]

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@unique([companyId, documentNo])
  @@index([companyId])
  @@index([categoryId])
  @@index([documentTypeId])
  @@index([customerId])
  @@index([employeeId])
  @@index([status])
  @@index([createdAt])
}
